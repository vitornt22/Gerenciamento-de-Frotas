    autoindex on;
    alias /root/app_repo/static;
  }

  # ATTENTION: /root/app_repo/media 
  location /media {
    autoindex on;
    alias /root/app_repo/media;
  }

  # ATTENTION: optar
  location / {
    proxy_pass http://unix:/run/optar.sock;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection 'upgrade';
    proxy_set_header Host $host;
    proxy_cache_bypass $http_upgrade;
  }

  # deny access to .htaccess files, if Apache's document root

  location ~ /\.ht {
    deny all;
  }

  location ~ /\. {
    access_log off;
    log_not_found off;
    deny all;
  }

  gzip_disable "msie6";

  gzip_comp_level 6;
  gzip_min_length 1100;
  gzip_buffers 4 32k;
  gzip_proxied any;
  gzip_types

    text/css
    text/js
    text/xml
    text/javascript

    application/x-javascript
    application/json
    application/xml
    application/rss+xml
